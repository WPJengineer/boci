-- Relationships
DROP TABLE IF EXISTS `boci_orders`,`boci_shopping_cart`,`boci_product_category`, `boci_reviews`; 
-- Entities
DROP TABLE IF EXISTS `boci_customers`,`boci_products`,`boci_categories`,`boci_address`;

DROP TABLE IF EXISTS `boci_customers`;
CREATE TABLE IF NOT EXISTS `boci_customers` (
  `customer_id` int(11) NOT NULL AUTO_INCREMENT,
  `customer_forename` varchar(100) NOT NULL,
  `customer_lastname` varchar(100) NOT NULL,
  `customer_role` enum('guest','customer','admin','') DEFAULT 'guest',
  -- `customer_username` varchar(100) UNIQUE DEFAULT NULL,
  `customer_email` varchar(300) UNIQUE NOT NULL,
  `customer_password` varchar(100) NOT NULL,
  `customer_gender` varchar(100) NOT NULL,
  `customer_privacy` TINYINT(1) NOT NULL DEFAULT 1,
  `customer_newsletter` TINYINT(1) DEFAULT 1,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`customer_id`)
);

DROP TABLE IF EXISTS `boci_products`;
CREATE TABLE IF NOT EXISTS `boci_products` (
  `product_id` int(11) NOT NULL,
  `product_name` varchar(300) DEFAULT NULL,
  `product_unit_price` decimal(10,2) DEFAULT NULL,
  `product_image` varchar(300) DEFAULT NULL,
  `product_perspective_1` varchar(300) DEFAULT NULL,
  `product_perspective_2` varchar(300) DEFAULT NULL,
  `product_reference` varchar(300) DEFAULT NULL,
  `product_rating` decimal(3,2) DEFAULT 5.00,
  `product_description` varchar(300) DEFAULT '',
  `product_stock` int(11) DEFAULT 0,
  `isStock` TINYINT(1) NOT NULL DEFAULT 0,
  PRIMARY KEY (`product_id`)
);

DROP TABLE IF EXISTS `boci_reviews`;
CREATE TABLE IF NOT EXISTS `boci_reviews` (
  `review_id` int(11) NOT NULL AUTO_INCREMENT,
  `product_id` int(11) DEFAULT NULL,
  `customer_id` int(11) DEFAULT NULL,
  `review_score` int(11) DEFAULT NULL,
  `review_content` varchar(300) DEFAULT NULL,
  PRIMARY KEY (`review_id`),
  KEY `product_id` (`product_id`),
  KEY `customer_id` (`customer_id`)
);

DROP TABLE IF EXISTS `boci_shopping_cart`;
CREATE TABLE IF NOT EXISTS `boci_shopping_cart` (
  `customer_id` int(11) NOT NULL,
  `product_id` int(11) NOT NULL,
  `quantity` int(11) DEFAULT NULL,
  `inserted_on` date DEFAULT current_timestamp(),
  PRIMARY KEY (`customer_id`,`product_id`),
  KEY `product_id` (`product_id`)
);

DROP TABLE IF EXISTS `boci_orders`;
CREATE TABLE IF NOT EXISTS `boci_orders` (
  `order_number` varchar(100) NOT NULL,
  `customer_id` int(11) NOT NULL,
  `product_id` int(11) NOT NULL,
  `quantity` int(11) DEFAULT NULL,
  `product_unit_price` decimal(10,2) DEFAULT NULL,
  `placed_on` date DEFAULT current_timestamp(),
  PRIMARY KEY (`order_number`,`customer_id`,`product_id`),
  KEY `customer_id` (`customer_id`),
  KEY `product_id` (`product_id`)
);

DROP TABLE IF EXISTS `boci_categories`;
CREATE TABLE IF NOT EXISTS `boci_categories` (
  `category_id` int(11) NOT NULL AUTO_INCREMENT,
  `category_name` varchar(300) DEFAULT NULL,
  PRIMARY KEY (`category_id`)
);

DROP TABLE IF EXISTS `boci_product_category`;
CREATE TABLE IF NOT EXISTS `boci_product_category` (
  `product_id` int(11) DEFAULT NULL,
  `category_id` int(11) DEFAULT NULL,
  KEY `product_id` (`product_id`),
  KEY `category_id` (`category_id`)
);

DROP TABLE IF EXISTS `boci_address`;
CREATE TABLE IF NOT EXISTS `boci_address` (
  `address_id` int(11) NOT NULL AUTO_INCREMENT,
  `customer_id` int(11) NOT NULL,
  `street` varchar(300) NOT NULL,
  `number` varchar(300) DEFAULT NULL,
  `city` varchar(300) NOT NULL,
  `state` varchar(300) NOT NULL,
  `postal_code` varchar(300) NOT NULL,
  `country` varchar(300) NOT NULL,
  PRIMARY KEY (`address_id`),
  KEY `customer_id` (`customer_id`)
);
